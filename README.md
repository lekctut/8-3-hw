код gitlab-ci.yml
```
stages:
  - test
  - build

test:
  stage: test
  image: golang:1.17
  script: 
   - go test .

build:
  stage: build
  image: docker:latest
  script:
   - docker build . -t go_app_v0.0.$CI_JOB_ID
```
